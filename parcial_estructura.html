Pseudocódigo

FUNCIÓN ConstruirGrafo(nodos, zonas_interferencia, viento):
    grafo = GRAFO_VACIO()
    
    // Agregar nodos
    PARA cada nodo en nodos:
        grafo.agregarNodo(nodo)
    
    // Calcular y agregar aristas
    PARA cada par (nodo_i, nodo_j) en nodos:
        distancia = calcularDistanciaEuclidiana(nodo_i, nodo_j)
        SI no hayInterseccionConZonas(nodo_i, nodo_j, zonas_interferencia):
            factor_viento = calcularFactorViento(nodo_i, nodo_j, viento)
            peso = distancia * (1 + factor_viento) * consumo_bateria_por_unidad
            grafo.agregarArista(nodo_i, nodo_j, peso)
    
    RETORNAR grafo

FUNCIÓN hayInterseccionConZonas(nodo_i, nodo_j, zonas):
    linea = segmento(nodo_i, nodo_j)
    PARA cada zona en zonas_interferencia:
        SI linea.intersecta(zona):
            RETORNAR VERDADERO
RETORNAR FALSO
