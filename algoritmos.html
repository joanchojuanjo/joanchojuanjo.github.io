Factorial de un Número

Inicio
    Definir n, factorial Como Entero
    Escribir "Ingrese un número: "
    Leer n
    factorial <- 1
    Para i <- 1 Hasta n Hacer
        factorial <- factorial * i
    FinPara
    Escribir "El factorial es: ", factorial
Fin

Quick Sort

Inicio Quicksort(lista, inicio, fin)
    Si inicio < fin Entonces
        pivote <- Particionar(lista, inicio, fin)
        Quicksort(lista, inicio, pivote - 1)
        Quicksort(lista, pivote + 1, fin)
    FinSi
Fin

Funcion Particionar(lista, inicio, fin)
    pivote <- lista[fin]
    i <- inicio - 1
    Para j <- inicio hasta fin - 1 Hacer
        Si lista[j] <= pivote Entonces
            i <- i + 1
            Intercambiar lista[i] con lista[j]
        FinSi
    FinPara
    Intercambiar lista[i + 1] con lista[fin]
    Retornar i + 1
FinFuncion

Merge Sort

Inicio Mergesort(lista)
    Si longitud(lista) <= 1 Entonces
        Retornar lista
    FinSi
    mitad <- longitud(lista) / 2
    izquierda <- Mergesort(sublista(lista, 0, mitad))
    derecha <- Mergesort(sublista(lista, mitad, longitud(lista)))
    Retornar Mezclar(izquierda, derecha)
Fin

Funcion Mezclar(izquierda, derecha)
    resultado <- []
    Mientras izquierda no esté vacía Y derecha no esté vacía Hacer
        Si izquierda[0] <= derecha[0] Entonces
            Agregar izquierda[0] a resultado
            Eliminar izquierda[0]
        Sino
            Agregar derecha[0] a resultado
            Eliminar derecha[0]
        FinSi
    FinMientras
    Concatenar resultado con izquierda y derecha
    Retornar resultado
FinFuncion

Implementación de BFS

Inicio BFS(grafo, nodo_inicial)
crear cola vacía
agregar nodo_inicial a la cola
marcar nodo_inicial como visitado
Mientras la cola no esté vacía Hacer
    nodo <- extraer de la cola
    Para cada vecino en lista_adyacencia(nodo) Hacer
        Si vecino no ha sido visitado Entonces
            marcar vecino como visitado
            agregar vecino a la cola
        FinSi
    FinPara
FinMientras
Fin

Busqueda Lineal y Binaria

Inicio BusquedaLineal(lista, objetivo)
    Para i <- 0 hasta longitud(lista) - 1 Hacer
        Si lista[i] == objetivo Entonces
            Retornar i
        FinSi
    FinPara
    Retornar -1
Fin

Inicio BusquedaBinaria(lista, objetivo)
    inicio <- 0
    fin <- longitud(lista) - 1
    Mientras inicio <= fin Hacer
        medio <- (inicio + fin) / 2
        Si lista[medio] == objetivo Entonces
            Retornar medio
        Sino Si lista[medio] < objetivo Entonces
            inicio <- medio + 1
        Sino
            fin <- medio - 1
        FinSi
    FinMientras
    Retornar -1
Fin
